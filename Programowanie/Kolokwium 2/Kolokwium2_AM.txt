double VectorProduct(double vector1[2], double vector2[2]) {
	return vector1[0]*vector2[0]+vector1[1]*vector2[1];
}

double textToNumber(char text[100]) {
	int len = 0, tmp = 0, power = 1;
	char tmpC;
	for (int i = 0; i < 100; i++) 
	{
		if(text[i]<'0'||text[i]>'9')
			break;
		len++;
	}
	for (int i = len-1; i >= 0; i--) {
		tmp+=((int)text[i] - (int)'0') * power;
		//printf("%c", text[i]);
		power*=10.;
	}
	return tmp;
}

int main()
{
	setlocale(LC_ALL, "");
	FILE * fRef;
	double tmp = 0., min = 0, max= 0, avg=0;
	char currentLine[100] = "";
	int isFirst = 1, len = 0;
	

	fRef = fopen("dane.txt", "r");
	if (fRef == NULL)
	{
		printf("Nie mo¿na otworzyæ pliku!");
		return 0;
	}

	while (fgets(currentLine, sizeof(currentLine), fRef) != NULL) {
		tmp = textToNumber(currentLine);
		if (isFirst) {
			min = tmp;
			max = tmp;
			avg = tmp;
			isFirst = 0;
		}
		else {
			if(min>tmp)

				max = tmp;
			avg+=tmp;
		}
