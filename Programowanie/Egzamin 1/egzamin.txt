void flushKeyBoard()
{
	int ch;
	while ((ch = getc(stdin)) != EOF && ch != '\n');
}

double CalculateConeVolume(double r, double l)
{
	double h;
	h = sqrt((l * l) - (r * r));
	return (M_PI * (r * r) * h) / 3.0;
}

double CalculateConeSurfaceArea(double r, double l)
{
	return M_PI * r * (r + l);
}

int maxIndex(double* x, int len)
{
	int index = 0;
	double max = 0.0;

	for (int i = 0; i < len; i++) {
		if (x[i] > max) {
			max = x[i];
			index = i+1;
		}
	}

	
	return index;
}
