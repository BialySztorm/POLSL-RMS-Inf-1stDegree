#include <stdio.h>
#include <conio.h>
#include <locale.h>
#pragma warning(disable : 4996)
#define _CRT_SECURE_NO_WARNINGS
#define _USE_MATH_DEFINES
#include <math.h>

int main()
{
	setlocale(LC_ALL, "");
	const int n = 10;
	int tab[n], min, max = 0;
	for (int i = 0; i < 10; i++) {
		printf("podaj %i liczbê: ", i+1);
		scanf("%i", &tab[i]);
	}

	min = tab[0];
	for (int i = 0; i < n; i++) {
		if (min > tab[i])
			min = tab[i];
		if (max < tab[i])
			max = tab[i];
	}

	printf("min: %d, max: %d", min, max);

	_getch();
}

#include <stdio.h>
#include <conio.h>
#include <locale.h>
#pragma warning(disable : 4996)
#define _CRT_SECURE_NO_WARNINGS
#define _USE_MATH_DEFINES
#include <math.h>

int main()
{
	setlocale(LC_ALL, "");
	const int n = 10;
	int tab[n], tmp;
	for (int i = 0; i < 10; i++) {
		printf("podaj %i liczbê: ", i + 1);
		scanf("%i", &tab[i]);
	}

	for (int i = n; i > 1; i--) {
		for (int j = 0; j < i; j++) {
			if (tab[j] > tab[j + 1]) {
				tmp = tab[j];
				tab[j] = tab[j + 1];
				tab[j + 1] = tmp;
			}
		}
	}

	for (int i = 0; i < 10; i++) {
		printf("%d, ", tab[i]);
	}

	_getch();
}
