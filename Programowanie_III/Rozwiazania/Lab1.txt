import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Podaj warto콑캖 洧논: ");
        double x = scanner.nextDouble();
        System.out.print("Podaj warto콑캖 洧녩: ");
        double a = scanner.nextDouble();
        System.out.print("Podaj warto콑캖 洧녪: ");
        double b = scanner.nextDouble();
        System.out.print("Podaj warto콑캖 洧녫: ");
        double c = scanner.nextDouble();
        System.out.print("Podaj warto콑캖 洧녬: ");
        double d = scanner.nextDouble();
        double result1 = horner(x, a, b, c);
        double result2 = horner(x, a, b, c, d);
        System.out.println("Wynik dla 洧냢(洧논) = 洧녩洧논^2 + 洧녪洧논 + 洧녫: " + result1);
        System.out.println("Wynik dla 洧냣(洧논) = 洧녩洧논^3 + 洧녪洧논^2 + 洧녫洧논 + 洧녬: " + result2);
    }

    public static double horner(double x, double... coefficients) {
        double result = coefficients[coefficients.length - 1];
        for (int i = coefficients.length - 2; i >= 0; i--) {
            result = result * x + coefficients[i];
        }
        return result;
    }
}
